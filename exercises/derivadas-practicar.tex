\begin{exercise}
  Obtener la derivada de las siguientes funciones:
  \begin{enumeratealpha}
    \item $f(x)=2x^8-3x^5+5$
    \item $f(x)=3x^{4/3}-6x^{2/3}-2$
    \item $f(x)=\sqrt{x}-\dfrac{1}{\sqrt{x}}$
    \item $f(x)=\sqrt[3]{x^2}+2\sqrt{x^3}$
    \item $f(x)=3x+\dfrac{1}{x}-\dfrac{2}{x^2}+\dfrac{3}{\sqrt[3]{x}}$
    \item $f(x)=\dfrac{2x+5}{3x-2}$
    \item $f(x)=\dfrac{x^2}{1-x^3}$
    \item $f(x)= \dfrac{(x+1)(x+2)}{(x-1)(x-2)}$
    \item $f(x)= \sqrt[3]{x}\ln x$
    \item $f(x)= \left(2e^x+x^2\right)\ln x$
    \item $f(x)= \dfrac{1-x^2}{1+x^2}$
    \item $f(x)= \dfrac{\ln x}{x+1}$
    \item $f(x)= \dfrac{1+\ln x}{1-\ln x}$
    \item $f(x)= \left(x^3+3x^2-5x-3\right)^5$
    \item $f(x)= \sqrt{x^2-8x+1}$
    \item $f(x)= \dfrac{1}{\left(x^2-2x-5\right)^4}$
    \item $f(x)= \dfrac{1}{\sqrt[5]{2x^2+3x-1}}$
    \item $f(x)= \sqrt[4]{\dfrac{x^3+1}{x^2-1}}$
    \item $f(x)= \sqrt{\ln x}$
    \item $f(x)= \ln \sqrt{\dfrac{x-1}{x+1}}$
  \end{enumeratealpha}
\end{exercise}

\begin{solution}
  \begin{enumeratealpha}
    \item $f'(x)=16x^7-15x^4$
    \item $f'(x)=-4x^{-1/3}+4x^{1/3}$
    \item $f'(x)=\dfrac{1}{2\sqrt{x}}+\dfrac{1}{2x\sqrt{x}}$
    \item $f'(x)=\dfrac{2}{3\sqrt[3]{x}}+3\sqrt{x}$
    \item $f'(x)=3-\dfrac{1}{x^2}+\dfrac{4}{x^3}-\dfrac{1}{\sqrt[3]{x^4}}$
    \item $f'(x)=\dfrac{-19}{(3x-2)^2}$
    \item $f'(x)=\dfrac{x^4+2x}{x^6-2x^3+1}$
    \item $f'(x)=\dfrac{-6x^2+12}{(x-1)^2(x-2)^2}$
    \item $f'(x)=\dfrac{3+\ln x}{3\sqrt[3]{x^2}}$
    \item $f'(x)=2e^x\ln x+\dfrac{2e^x}{x}+2x\ln x+x$
    \item $f'(x)=\dfrac{-4x}{\left(1+x^2\right)^2}$
    \item $f'(x)=\dfrac{1}{x(x+1)}-\dfrac{\ln x}{(x+1)^2}$
    \item $f'(x)=\dfrac{2}{x\left(\ln^2x-2\ln x+1\right)}$
    \item $f'(x)=5\left(3x^2+6x-5\right)\left(x^3+3x^2-5x-3\right)^4$
    \item $f'(x)=\dfrac{x-4}{\sqrt{x^2-8x+1}}$
    \item $f'(x)=\dfrac{-8\left(x-1\right)}{\left(x^2-2x-5\right)^5}$
    \item $f'(x)=\dfrac{-4x-3}{5\sqrt[5]{\left(2x^2+3x-1\right)^6}}$
    \item $f'(x)=\dfrac{x(x-2)}{4(x-1)^2\sqrt[4]{\left(\dfrac{x^2-x+1}{x-1}\right)^3}}$
    \item $f'(x)=\dfrac{1}{2x\sqrt{\ln x}}$
    \item $f'(x)=\dfrac{1}{x^2-1}$
  \end{enumeratealpha}
\end{solution}

\begin{exercise}
  Obtener la derivada de las siguientes funciones:
  \begin{enumeratealpha}
    \item $f(x)= \thpwf{-x^2}{x\leq0}{x^3}{0<x<1}{2}{x\geq1}$
    \item $f(x)= x^2-|x|$
    \item $f(x)=\sen e^x$
    \item $f(x)=\arctg\left(\sqrt{x}-x\right)$
    \item $f(x)=\ln\left(\ln\left(\ln x\right)\right)$
    \item $f(x)=\dfrac{\sen 2x}{1+\cos^2x}$
    \item $f(x)=\sen\left(\ln|x|\right)$
    \item $f(x)=\ln\left|\sen x\right|$
    \item $f(x)=x^{\sen x}$
    \item $f(x)=\left(\cos(2x)\right)^{1/x^2}$
   \end{enumeratealpha}
\end{exercise}

\begin{solution}
  \begin{enumeratealpha}
    \item $f'(x)=\thpwf{-2x}{x\leq0}{3x^2}{0<x<1}{0}{x>1}$
    \item $f'(x)=2x-\dfrac{|x|}{x}=\twpwf{2x+1}{x<0}{2x-1}{x>0}$
    \item $f'(x)=e^x\sen e^x$
    \item $f'(x)=\dfrac{\frac{1}{2}-\sqrt{x}}{\sqrt[5]{x}+\sqrt[3]{x}+\sqrt{x}-2x^2}$
    \item $f'(x)=\dfrac{1}{x\left(\ln x\right)\ln\left(\ln x\right)}$
    \item $f'(x)=\dfrac{6\cos^2x-2}{\left(1+\cos^2x\right)^2}$
    \item $f'(x)=\dfrac{1}{x}\cos\left(\ln|x|\right)$
    \item $f'(x)=\cotg x$
    \item $f'(x)=x^{\sen x}\left((\ln x)\cos x+\dfrac{1}{x}\sen x\right)$
    \item $f'(x)=\dfrac{-2\left(\cos(2x)\right)^{1/x^2}}{x^3}\left(\ln\left(\cos(2x)\right)+x\tg(2x)\right)$
  \end{enumeratealpha}
\end{solution}

\begin{exercise}
  Halla las cuatro primeras derivadas de cada una de las funciones siguientes:
  \begin{enumeratealpha}
    \item $f(x)= x^4+x^3-2x^2+x-5$
    \item $f(x)= x^2+x+3$
    \item $f(x)= \sen^2 x$
    \item $f(x)= \dfrac{x}{x+1}$
  \end{enumeratealpha}
\end{exercise}

\begin{solution}
  \begin{enumeratealpha}
    \item $f'(x)=4x^3+3x^2-4x+1$; $f''(x)=12x^2+6x-4$; $f'''(x)=24x+6$; $f^{(4)}(x)=24$.
    \item $f'(x)=2x+1$; $f''(x)=2$; $f'''(x)=f^{(4)}(x)=0$.
    \item $f'(x)=2\sen x\cos x=\sen(2x);\ f''(x)=2\cos^2x-2\sen^2x=2\cos(2x);\ f'''(x)=-8\sen x\cos x=-4\sen(2x);\ f^{(4)}(x)=-8\cos^2x-8\sen^2x=-8\cos(2x)$.
    \item $f'(x)=\dfrac{1}{(x+1)^2}$; $f''(x)=\dfrac{-2}{(x+1)^3}$; $f'''(x)=\dfrac{6}{(x+1)^4}$; $f^{(4)}(x)=\dfrac{-24}{(x+1)^5}$
  \end{enumeratealpha}
\end{solution}
